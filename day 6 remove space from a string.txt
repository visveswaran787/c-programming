#include <stdio.h>
#include <string.h>
void removeSpaces(char *str) {
    int i, j;
    j = 0; 
    for (i = 0; str[i] != '\0'; i++) {
        if (str[i] != ' ') {
            str[j] = str[i];
            j++;
        }
    }
    str[j] = '\0'; 
}
int main() {
    char text[100];
    printf("Enter a string: ");
    fgets(text, sizeof(text), stdin);
    text[strcspn(text, "\n")] = '\0';
    printf("Original string: \"%s\"\n", text);
    removeSpaces(text);
    printf("String after removing spaces: \"%s\"\n", text);
    return 0;
}
