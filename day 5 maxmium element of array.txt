#include <stdio.h>
int main() {
    int arr[5]; // Declare an array of size 5
    int i;
    int max_element;
    printf("Enter 5 integer elements for the array:\n");
    for (i = 0; i < 5; i++) {
        printf("Enter element %d: ", i + 1);
        scanf("%d", &arr[i]);
    }
    max_element = arr[0];
    for (i = 1; i < 5; i++) {
        if (arr[i] > max_element) {
            max_element = arr[i];
        }
    }
    printf("\nThe maximum element in the array is: %d\n", max_element);
    return 0;
}
