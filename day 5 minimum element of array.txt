#include <stdio.h>
int main() {
    int arr[] = {5, 2, 8, 2, 9, 3}; 
    int n = sizeof(arr) / sizeof(arr[0]);
    int min_element;
    if (n == 0) {
        printf("The array is empty.\n");
        return 1; 
    }
    min_element = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] < min_element) {
            min_element = arr[i];
        }
    }
    printf("The minimum element in the array is: %d\n", min_element);
    return 0; 
}
