#include <stdio.h>
int main() {
    char str[100];
    char ch;       
    int count = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); 
    printf("Enter the character to find its frequency: ");
    scanf(" %c", &ch); 
    for (int i = 0; str[i] != '\0'; ++i) {
        if (ch == str[i]) {
            count++; 
        }
    }
    printf("Frequency of '%c' in the string is: %d\n", ch, count);
    return 0; 
}
