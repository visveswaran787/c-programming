#include <stdio.h>
#include <stdlib.h>
int* mergeArrays(int arr1[], int n1, int arr2[], int n2) {
    int *mergedArr = (int *)malloc((n1 + n2) * sizeof(int));
    for (int i = 0; i < n1; i++) {
        mergedArr[i] = arr1[i];
    }
    for (int i = 0; i < n2; i++) {
        mergedArr[n1 + i] = arr2[i];
    }
    return mergedArr;
}
int main() {
    int arr1[] = {10, 20, 30};
    int n1 = sizeof(arr1) / sizeof(arr1[0]);
    int arr2[] = {40, 50, 60, 70};
    int n2 = sizeof(arr2) / sizeof(arr2[0]); 
    int *resultArray = mergeArrays(arr1, n1, arr2, n2);
    printf("Merged Array: ");
    for (int i = 0; i < (n1 + n2); i++) {
        printf("%d ", resultArray[i]);
    }
    printf("\n");
    free(resultArray);
    return 0;
}
