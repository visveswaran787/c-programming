#include <stdio.h>
void toUpperCase(char *str) {
    int i = 0;
    while (str[i] != '\0') {
        if (str[i] >= 'a' && str[i] <= 'z') {
            str[i] = str[i] - 32; 
        }
        i++;
    }
}
void toLowerCase(char *str) {
    int i = 0;
    while (str[i] != '\0') {
        if (str[i] >= 'A' && str[i] <= 'Z') {
            str[i] = str[i] + 32; 
        }
        i++;
    }
}
int main() {
    char myString[100];
    printf("Enter a string: ");
    fgets(myString, sizeof(myString), stdin); 
    int i = 0;
    while (myString[i] != '\0') {
        if (myString[i] == '\n') {
            myString[i] = '\0';
            break;
        }
        i++;
    }
    printf("Original string: %s\n", myString);
    toUpperCase(myString);
    printf("Uppercase string: %s\n", myString);
    printf("\nEnter the string again for lowercase conversion: ");
    fgets(myString, sizeof(myString), stdin); 
    i = 0;
    while (myString[i] != '\0') {
        if (myString[i] == '\n') {
            myString[i] = '\0';
            break;
        }
        i++;
    }
    toLowerCase(myString);
    printf("Lowercase string: %s\n", myString);
    return 0;
}
