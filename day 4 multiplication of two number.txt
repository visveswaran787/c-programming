#include <stdio.h>
int main() {
    int r1, c1, r2, c2; 
    int i, j, k;        
    printf("Enter number of rows for First Matrix: ");
    scanf("%d", &r1);
    printf("Enter number of columns for First Matrix: ");
    scanf("%d", &c1);
    printf("Enter number of rows for Second Matrix: ");
    scanf("%d", &r2);
    printf("Enter number of columns for Second Matrix: ");
    scanf("%d", &c2);
    if (c1 != r2) {
        printf("Matrices cannot be multiplied. Number of columns of first matrix must equal number of rows of second matrix.\n");
        return 1; 
    }
    int m1[r1][c1];
    int m2[r2][c2];
    int mul[r1][c2]; 
    printf("\nEnter elements of the first matrix:\n");
    for (i = 0; i < r1; i++) {
        for (j = 0; j < c1; j++) {
            printf("Enter element m1[%d][%d]: ", i, j);
            scanf("%d", &m1[i][j]);
        }
    }
    printf("\nEnter elements of the second matrix:\n");
    for (i = 0; i < r2; i++) {
        for (j = 0; j < c2; j++) {
            printf("Enter element m2[%d][%d]: ", i, j);
            scanf("%d", &m2[i][j]);
        }
    }
    for (i = 0; i < r1; i++) {
        for (j = 0; j < c2; j++) {
            mul[i][j] = 0; // Initialize element of resultant matrix to 0
            for (k = 0; k < c1; k++) {
                mul[i][j] += m1[i][k] * m2[k][j];
            }
        }
    }
    printf("\nResultant matrix after multiplication:\n");
    for (i = 0; i < r1; i++) {
        for (j = 0; j < c2; j++) {
            printf("%d\t", mul[i][j]);
        }
        printf("\n");
    }

    return 0; 
}
